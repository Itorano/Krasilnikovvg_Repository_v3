#include <iostream>
using namespace std;

int Input_Digit(int Digit) {
    cout << "2. Введите цифру A: ";
    cin >> Digit;
    return Digit;
}

int main() {
    int X, A, Command;

    do
    {
        cout << "Введите номер команды:";
        cin >> Command;
        
        switch (Command) {
            case 1: 
                X = Input_Number(X); 
                break;
            case 2: 
                A = Input_Digit(A);
                break;
            case 3:
                if (Digit_Presence(X, A)) {
                    cout << "3. Цифра " << A << " отсутствует в числе " << X << endl;
                } else {
                    cout << "3. Цифра " << A << " присутствует в числе " << X << endl;
                }
                break;
            case 4:
                if (Digits_More_Than_Twice(X, A)) {
                    cout << "4. Цифра " << A << " встречается более двух раз в числе " << X << endl;
                } else {
                    cout << "4. Цифра " << A << " встречается не более двух раз в числе " << X << endl;
                }
                break;
        }
    } while(Command > 0 && Command < 5);

    return 0;
}
